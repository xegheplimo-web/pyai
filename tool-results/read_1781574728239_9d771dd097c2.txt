   109→def _coerce_request_bool(value: Any, default: bool = False) -> bool:
   110→    """Normalize boolean-like API payload values.
   111→
   112→    External clients should send real JSON booleans, but some OpenAI-compatible
   113→    frontends and middleware serialize flags like ``stream`` as strings.  Using
   114→    Python truthiness on those values misroutes requests because ``"false"`` is
   115→    still truthy.  Treat only explicit bool-ish scalars as booleans; everything
   116→    else falls back to the caller's default.
   117→    """
   118→    if isinstance(value, bool):
   119→        return value
   120→    if value is None:
   121→        return default
   122→    if isinstance(value, str):
   123→        normalized = value.strip().lower()
   124→        if normalized in _TRUE_REQUEST_BOOL_STRINGS:
   125→            return True
   126→        if normalized in _FALSE_REQUEST_BOOL_STRINGS:
   127→            return False
   128→        return default
   129→    if isinstance(value, (int, float)):
   130→        return bool(value)
   131→    return default

   134→def _normalize_chat_content(
   135→    content: Any, *, _max_depth: int = 10, _depth: int = 0,
   136→) -> str:
   137→    """Normalize OpenAI chat message content into a plain text string.
   138→
   139→    Some clients (Open WebUI, LobeChat, etc.) send content as an array of
   140→    typed parts instead of a plain string::
   141→
   142→        [{"type": "text", "text": "hello"}, {"type": "input_text", "text": "..."}]
   143→
   144→    This function flattens those into a single string so the agent pipeline
   145→    (which expects strings) doesn't choke.
   146→
   147→    Defensive limits prevent abuse: recursion depth, list size, and output
   148→    length are all bounded.
   149→    """
   150→    if _depth > _max_depth:
   151→        return ""
   152→    if content is None:
   153→        return ""
   154→    if isinstance(content, str):
   155→        return content[:MAX_NORMALIZED_TEXT_LENGTH] if len(content) > MAX_NORMALIZED_TEXT_LENGTH else content
   156→
   157→    if isinstance(content, list):
   158→        parts: List[str] = []
   159→        total_len = 0
   160→        items = content[:MAX_CONTENT_LIST_SIZE] if len(content) > MAX_CONTENT_LIST_SIZE else content
   161→        for item in items:
   162→            if isinstance(item, str):
   163→                if item:
   164→                    part = item[:MAX_NORMALIZED_TEXT_LENGTH]
   165→                    parts.append(part)
   166→                    total_len += len(part)
   167→            elif isinstance(item, dict):
   168→                item_type = str(item.get("type") or "").strip().lower()
   169→                if item_type in {"text", "input_text", "output_text"}:
   170→                    text = item.get("text", "")
   171→                    if text:
   172→                        try:
   173→                            part = str(text)[:MAX_NORMALIZED_TEXT_LENGTH]
   174→                            parts.append(part)
   175→                            total_len += len(part)
   176→                        except Exception:
   177→                            pass
   178→                # Silently skip image_url / other non-text parts
   179→            elif isinstance(item, list):
   180→                nested = _normalize_chat_content(item, _max_depth=_max_depth, _depth=_depth + 1)
   181→                if nested:
   182→                    parts.append(nested)
   183→                    total_len += len(nested)
   184→            # Check accumulated size
   185→            if total_len >= MAX_NORMALIZED_TEXT_LENGTH:
   186→                break
   187→        result = "\n".join(parts)
   188→        return result[:MAX_NORMALIZED_TEXT_LENGTH] if len(result) > MAX_NORMALIZED_TEXT_LENGTH else result
   189→
   190→    # Fallback for unexpected types (int, float, bool, etc.)
   191→    try:
   192→        result = str(content)
   193→        return result[:MAX_NORMALIZED_TEXT_LENGTH] if len(result) > MAX_NORMALIZED_TEXT_LENGTH else result
   194→    except Exception:
   195→        return ""
   196→
   197→
   198→# Content part type aliases used by the OpenAI Chat Completions and Responses
   199→# APIs.  We accept both spellings on input and emit a single canonical internal
   200→# shape (``{"type": "text", ...}`` / ``{"type": "image_url", ...}``) that the
   201→# rest of the agent pipeline already understands.
   202→_TEXT_PART_TYPES = frozenset({"text", "input_text", "output_text"})
   203→_IMAGE_PART_TYPES = frozenset({"image_url", "input_image"})
   204→_FILE_PART_TYPES = frozenset({"file", "input_file"})

   207→def _normalize_multimodal_content(content: Any) -> Any:
   208→    """Validate and normalize multimodal content for the API server.
   209→
   210→    Returns a plain string when the content is text-only, or a list of
   211→    ``{"type": "text"|"image_url", ...}`` parts when images are present.
   212→    The output shape is the native OpenAI Chat Completions vision format,
   213→    which the agent pipeline accepts verbatim (OpenAI-wire providers) or
   214→    converts (``_preprocess_anthropic_content`` for Anthropic).
   215→
   216→    Raises ``ValueError`` with an OpenAI-style code on invalid input:
   217→      * ``unsupported_content_type`` — file/input_file/file_id parts, or
   218→        non-image ``data:`` URLs.
   219→      * ``invalid_image_url`` — missing URL or unsupported scheme.
   220→      * ``invalid_content_part`` — malformed text/image objects.
   221→
   222→    Callers translate the ValueError into a 400 response.
   223→    """
   224→    # Scalar passthrough mirrors ``_normalize_chat_content``.
   225→    if content is None:
   226→        return ""
   227→    if isinstance(content, str):
   228→        return content[:MAX_NORMALIZED_TEXT_LENGTH] if len(content) > MAX_NORMALIZED_TEXT_LENGTH else content
   229→    if not isinstance(content, list):
   230→        # Mirror the legacy text-normalizer's fallback so callers that
   231→        # pre-existed image support still get a string back.
   232→        return _normalize_chat_content(content)
   233→
   234→    items = content[:MAX_CONTENT_LIST_SIZE] if len(content) > MAX_CONTENT_LIST_SIZE else content
   235→    normalized_parts: List[Dict[str, Any]] = []
   236→    text_accum_len = 0
   237→
   238→    for part in items:
   239→        if isinstance(part, str):
   240→            if part:
   241→                trimmed = part[:MAX_NORMALIZED_TEXT_LENGTH]
   242→                normalized_parts.append({"type": "text", "text": trimmed})
   243→                text_accum_len += len(trimmed)
   244→            continue
   245→
   246→        if not isinstance(part, dict):
   247→            # Ignore unknown scalars for forward compatibility with future
   248→            # Responses API additions (e.g. ``refusal``).  The same policy
   249→            # the text normalizer applies.
   250→            continue
   251→
   252→        raw_type = part.get("type")
   253→        part_type = str(raw_type or "").strip().lower()
   254→
   255→        if part_type in _TEXT_PART_TYPES:
   256→            text = part.get("text")
   257→            if text is None:
   258→                continue
   259→            if not isinstance(text, str):
   260→                text = str(text)
   261→            if text:
   262→                trimmed = text[:MAX_NORMALIZED_TEXT_LENGTH]
   263→                normalized_parts.append({"type": "text", "text": trimmed})
   264→                text_accum_len += len(trimmed)
   265→            continue
   266→
   267→        if part_type in _IMAGE_PART_TYPES:
   268→            detail = part.get("detail")
   269→            image_ref = part.get("image_url")
   270→            # OpenAI Responses sends ``input_image`` with a top-level
   271→            # ``image_url`` string; Chat Completions sends ``image_url`` as
   272→            # ``{"url": "...", "detail": "..."}``.  Support both.
   273→            if isinstance(image_ref, dict):
   274→                url_value = image_ref.get("url")
   275→                detail = image_ref.get("detail", detail)
   276→            else:
   277→                url_value = image_ref
   278→            if not isinstance(url_value, str) or not url_value.strip():
   279→                raise ValueError("invalid_image_url:Image parts must include a non-empty image URL.")
   280→            url_value = url_value.strip()
   281→            lowered = url_value.lower()
   282→            if lowered.startswith("data:"):
   283→                if not lowered.startswith("data:image/") or "," not in url_value:
   284→                    raise ValueError(
   285→                        "unsupported_content_type:Only image data URLs are supported. "
   286→                        "Non-image data payloads are not supported."
   287→                    )
   288→            elif not (lowered.startswith("http://") or lowered.startswith("https://")):
   289→                raise ValueError(
   290→                    "invalid_image_url:Image inputs must use http(s) URLs or data:image/... URLs."
   291→                )
   292→            image_part: Dict[str, Any] = {"type": "image_url", "image_url": {"url": url_value}}
   293→            if detail is not None:
   294→                if not isinstance(detail, str) or not detail.strip():
   295→                    raise ValueError("invalid_content_part:Image detail must be a non-empty string when provided.")
   296→                image_part["image_url"]["detail"] = detail.strip()
   297→            normalized_parts.append(image_part)
   298→            continue
   299→
   300→        if part_type in _FILE_PART_TYPES:
   301→            raise ValueError(
   302→                "unsupported_content_type:Inline image inputs are supported, "
   303→                "but uploaded files and document inputs are not supported on this endpoint."
   304→            )
   305→
   306→        # Unknown part type — reject explicitly so clients get a clear error
   307→        # instead of a silently dropped turn.
   308→        raise ValueError(
   309→            f"unsupported_content_type:Unsupported content part type {raw_type!r}. "
   310→            "Only text and image_url/input_image parts are supported."
   311→        )
   312→
   313→    if not normalized_parts:
   314→        return ""
   315→
   316→    # Text-only: collapse to a plain string so downstream logging/trajectory
   317→    # code sees the native shape and prompt caching on text-only turns is
   318→    # unaffected.
   319→    if all(p.get("type") == "text" for p in normalized_parts):
   320→        return "\n".join(p["text"] for p in normalized_parts if p.get("text"))
   321→
   322→    return normalized_parts

   325→def _content_has_visible_payload(content: Any) -> bool:
   326→    """True when content has any text or image attachment.  Used to reject empty turns."""
   327→    if isinstance(content, str):
   328→        return bool(content.strip())
   329→    if isinstance(content, list):
   330→        for part in content:
   331→            if isinstance(part, dict):
   332→                ptype = str(part.get("type") or "").strip().lower()
   333→                if ptype in _TEXT_PART_TYPES and str(part.get("text") or "").strip():
   334→                    return True
   335→                if ptype in _IMAGE_PART_TYPES:
   336→                    return True
   337→    return False

   341→    """Translate a ``_normalize_multimodal_content`` ValueError into a 400 response."""
   342→    raw = str(exc)
   343→    code, _, message = raw.partition(":")

   345→        code, message = "invalid_content_part", raw

   347→        _openai_error(message, code=code, param=param),
   348→        status=400,

   352→def _session_chat_user_message(body: Dict[str, Any], *, param: str = "message") -> tuple[Any, Optional["web.Response"]]:
   353→    """Parse and normalize session chat ``message`` / ``input`` like chat completions."""
   354→    user_message = body.get("message") or body.get("input")
   355→    if not _content_has_visible_payload(user_message):
   356→        return None, web.json_response(
   357→            _openai_error("Missing 'message' field", code="missing_message"),
   358→            status=400,
   359→        )
   360→    try:
   361→        return _normalize_multimodal_content(user_message), None
   362→    except ValueError as exc:
   363→        return None, _multimodal_validation_error(exc, param=param)

   371→class ResponseStore:
   372→    """
   373→    SQLite-backed LRU store for Responses API state.
   374→
   375→    Each stored response includes the full internal conversation history
   376→    (with tool calls and results) so it can be reconstructed on subsequent
   377→    requests via previous_response_id.
   378→
   379→    Persists across gateway restarts.  Falls back to in-memory SQLite
   380→    if the on-disk path is unavailable.
   381→    """
   382→
   383→    def __init__(self, max_size: int = MAX_STORED_RESPONSES, db_path: str = None):
   384→        self._max_size = max_size
   385→        if db_path is None:
   386→            try:
   387→                from hermes_cli.config import get_hermes_home
   388→                db_path = str(get_hermes_home() / "response_store.db")
   389→            except Exception:
   390→                db_path = ":memory:"
   391→        self._db_path: Optional[str] = db_path if db_path != ":memory:" else None
   392→        try:
   393→            self._conn = sqlite3.connect(db_path, check_same_thread=False)
   394→        except Exception:
   395→            self._conn = sqlite3.connect(":memory:", check_same_thread=False)
   396→            self._db_path = None
   397→        # Use shared WAL-fallback helper so response_store.db degrades
   398→        # gracefully on NFS/SMB/FUSE-mounted HERMES_HOME (same filesystem
   399→        # issue addressed for state.db/kanban.db — see
   400→        # hermes_state._WAL_INCOMPAT_MARKERS).
   401→        from hermes_state import apply_wal_with_fallback
   402→        apply_wal_with_fallback(self._conn, db_label="response_store.db")
   403→        self._conn.execute(
   404→            """CREATE TABLE IF NOT EXISTS responses (
   405→                response_id TEXT PRIMARY KEY,
   406→                data TEXT NOT NULL,
   407→                accessed_at REAL NOT NULL
   408→            )"""
   409→        )
   410→        self._conn.execute(
   411→            """CREATE TABLE IF NOT EXISTS conversations (
   412→                name TEXT PRIMARY KEY,
   413→                response_id TEXT NOT NULL
   414→            )"""
   415→        )
   416→        self._conn.commit()
   417→        # response_store.db contains conversation history (tool payloads,
   418→        # prompts, results). Tighten to owner-only after creation so other
   419→        # local users on a shared box can't read it. Run once at __init__
   420→        # rather than after every commit — chmod-on-every-write is wasted
   421→        # syscalls on a hot path.
   422→        self._tighten_file_permissions()
   423→
   424→    def _tighten_file_permissions(self) -> None:
   425→        """Force owner-only permissions on the DB and SQLite sidecars."""
   426→        if not self._db_path:
   427→            return
   428→        for candidate in (
   429→            Path(self._db_path),
   430→            Path(f"{self._db_path}-wal"),
   431→            Path(f"{self._db_path}-shm"),
   432→        ):
   433→            try:
   434→                if candidate.exists():
   435→                    candidate.chmod(0o600)
   436→            except OSError:
   437→                logger.debug(
   438→                    "Failed to restrict response store permissions for %s",
   439→                    candidate,
   440→                    exc_info=True,
   441→                )
   442→
   443→    def get(self, response_id: str) -> Optional[Dict[str, Any]]:
   444→        """Retrieve a stored response by ID (updates access time for LRU)."""
   445→        row = self._conn.execute(
   446→            "SELECT data FROM responses WHERE response_id = ?", (response_id,)
   447→        ).fetchone()
   448→        if row is None:
   449→            return None
   450→        self._conn.execute(
   451→            "UPDATE responses SET accessed_at = ? WHERE response_id = ?",
   452→            (time.time(), response_id),
   453→        )
   454→        self._conn.commit()
   455→        try:
   456→            return json.loads(row[0])
   457→        except (json.JSONDecodeError, TypeError):
   458→            logger.warning(
   459→                "Corrupted JSON in response store for id=%s, evicting entry",
   460→                response_id,
   461→            )
   462→            self._conn.execute(
   463→                "DELETE FROM responses WHERE response_id = ?",
   464→                (response_id,),
   465→            )
   466→            self._conn.commit()
   467→            return None
   468→
   469→    def put(self, response_id: str, data: Dict[str, Any]) -> None:
   470→        """Store a response, evicting the oldest if at capacity."""
   471→        self._conn.execute(
   472→            "INSERT OR REPLACE INTO responses (response_id, data, accessed_at) VALUES (?, ?, ?)",
   473→            (response_id, json.dumps(data, default=str), time.time()),
   474→        )
   475→        # Evict oldest entries beyond max_size
   476→        count = self._conn.execute("SELECT COUNT(*) FROM responses").fetchone()[0]
   477→        if count > self._max_size:
   478→            # Collect IDs that will be evicted
   479→            evict_ids = [
   480→                row[0]
   481→                for row in self._conn.execute(
   482→                    "SELECT response_id FROM responses ORDER BY accessed_at ASC LIMIT ?",
   483→                    (count - self._max_size,),
   484→                ).fetchall()
   485→            ]
   486→            if evict_ids:
   487→                placeholders = ",".join("?" for _ in evict_ids)
   488→                # Clear conversation mappings pointing to evicted responses
   489→                self._conn.execute(
   490→                    f"DELETE FROM conversations WHERE response_id IN ({placeholders})",
   491→                    evict_ids,
   492→                )
   493→                # Delete evicted responses
   494→                self._conn.execute(
   495→                    f"DELETE FROM responses WHERE response_id IN ({placeholders})",
   496→                    evict_ids,
   497→                )
   498→        self._conn.commit()
   499→
   500→    def delete(self, response_id: str) -> bool:
   501→        """Remove a response from the store. Returns True if found and deleted."""
   502→        # Clear conversation mappings pointing to this response
   503→        self._conn.execute(
   504→            "DELETE FROM conversations WHERE response_id = ?", (response_id,)
   505→        )
   506→        cursor = self._conn.execute(
   507→            "DELETE FROM responses WHERE response_id = ?", (response_id,)
   508→        )
   509→        self._conn.commit()
   510→        return cursor.rowcount > 0
   511→
   512→    def get_conversation(self, name: str) -> Optional[str]:
   513→        """Get the latest response_id for a conversation name."""
   514→        row = self._conn.execute(
   515→            "SELECT response_id FROM conversations WHERE name = ?", (name,)
   516→        ).fetchone()
   517→        return row[0] if row else None
   518→
   519→    def set_conversation(self, name: str, response_id: str) -> None:
   520→        """Map a conversation name to its latest response_id."""
   521→        self._conn.execute(
   522→            "INSERT OR REPLACE INTO conversations (name, response_id) VALUES (?, ?)",
   523→            (name, response_id),
   524→        )
   525→        self._conn.commit()
   526→
   527→    def close(self) -> None:
   528→        """Close the database connection."""
   529→        try:
   530→            self._conn.close()
   531→        except Exception:
   532→            pass
   533→
   534→    def __len__(self) -> int:
   535→        row = self._conn.execute("SELECT COUNT(*) FROM responses").fetchone()
   536→        return row[0] if row else 0

   625→class _IdempotencyCache:
   626→    """In-memory idempotency cache with TTL and basic LRU semantics."""
   627→    def __init__(self, max_items: int = 1000, ttl_seconds: int = 300):
   628→        from collections import OrderedDict
   629→        self._store = OrderedDict()
   630→        self._inflight: Dict[tuple[str, str], "asyncio.Task[Any]"] = {}
   631→        self._ttl = ttl_seconds
   632→        self._max = max_items
   633→
   634→    def _purge(self):
   635→        now = time.time()
   636→        expired = [k for k, v in self._store.items() if now - v["ts"] > self._ttl]
   637→        for k in expired:
   638→            self._store.pop(k, None)
   639→        while len(self._store) > self._max:
   640→            self._store.popitem(last=False)
   641→
   642→    async def get_or_set(self, key: str, fingerprint: str, compute_coro):
   643→        self._purge()
   644→        item = self._store.get(key)
   645→        if item and item["fp"] == fingerprint:
   646→            return item["resp"]
   647→
   648→        inflight_key = (key, fingerprint)
   649→        task = self._inflight.get(inflight_key)
   650→        if task is None:
   651→            async def _compute_and_store():
   652→                resp = await compute_coro()
   653→                import time as _t
   654→                self._store[key] = {"resp": resp, "fp": fingerprint, "ts": _t.time()}
   655→                self._purge()
   656→                return resp
   657→
   658→            task = asyncio.create_task(_compute_and_store())
   659→            self._inflight[inflight_key] = task
   660→
   661→            def _clear_inflight(done_task: "asyncio.Task[Any]") -> None:
   662→                if self._inflight.get(inflight_key) is done_task:
   663→                    self._inflight.pop(inflight_key, None)
   664→
   665→            task.add_done_callback(_clear_inflight)
   666→
   667→        return await asyncio.shield(task)
   668→
   669→
   670→_idem_cache = _IdempotencyCache()

   679→def _derive_chat_session_id(
   680→    system_prompt: Optional[str],
   681→    first_user_message: str,
   682→) -> str:
   683→    """Derive a stable session ID from the conversation's first user message.
   684→
   685→    OpenAI-compatible frontends (Open WebUI, LibreChat, etc.) send the full
   686→    conversation history with every request.  The system prompt and first user
   687→    message are constant across all turns of the same conversation, so hashing
   688→    them produces a deterministic session ID that lets the API server reuse
   689→    the same Hermes session (and therefore the same Docker container sandbox
   690→    directory) across turns.
   691→    """
   692→    seed = f"{system_prompt or ''}\n{first_user_message}"
   693→    digest = hashlib.sha256(seed.encode("utf-8")).hexdigest()[:16]
   694→    return f"api-{digest}"

   897→    def _check_auth(self, request: "web.Request") -> Optional["web.Response"]:
   898→        """
   899→        Validate Bearer token from Authorization header.
   900→
   901→        Returns None if auth is OK, or a 401 web.Response on failure.
   902→        connect() refuses to start the API server without API_SERVER_KEY, so
   903→        the no-key branch only exists for tests or unsupported manual wiring.
   904→        """
   905→        if not self._api_key:
   906→            return None
   907→
   908→        auth_header = request.headers.get("Authorization", "")
   909→        if auth_header.startswith("Bearer "):
   910→            token = auth_header[7:].strip()
   911→            if hmac.compare_digest(token, self._api_key):
   912→                return None  # Auth OK
   913→
   914→        logger.warning(
   915→            "API server rejected invalid API key: %s",
   916→            self._request_audit_log_suffix(request),
   917→        )
   918→        return web.json_response(
   919→            {"error": {"message": "Invalid API key", "type": "invalid_request_error", "code": "invalid_api_key"}},
   920→            status=401,
   921→        )

   923→    # ------------------------------------------------------------------
   924→    # Session header helpers
   925→    # ------------------------------------------------------------------
   926→
   927→    # Soft length cap for session identifiers.  Headers are bounded in
   928→    # aggregate by aiohttp (``client_max_size`` / default 8 KiB per
   929→    # header), but we impose a tighter limit on the session headers so a
   930→    # caller can't burn memory by passing a multi-kilobyte "session key".
   931→    # 256 chars is well above any realistic stable channel identifier
   932→    # (e.g. ``agent:main:webui:dm:user-42``) while staying small enough
   933→    # that the sanitized form is safe to pass into Honcho / state.db.
   934→    _MAX_SESSION_HEADER_LEN = 256
   935→
   936→    def _parse_session_key_header(
   937→        self, request: "web.Request"
   938→    ) -> tuple[Optional[str], Optional["web.Response"]]:
   939→        """Extract and validate the ``X-Hermes-Session-Key`` header.
   940→
   941→        The session key is a stable per-channel identifier that scopes
   942→        long-term memory (e.g. Honcho sessions) across transcripts.  It
   943→        is independent of ``X-Hermes-Session-Id``: callers may send
   944→        either, both, or neither.
   945→
   946→        Returns ``(session_key, None)`` on success (with an empty/absent
   947→        header yielding ``None`` for the key), or ``(None, error_response)``
   948→        on validation failure.
   949→
   950→        Security: like session continuation, accepting a caller-supplied
   951→        memory scope requires API-key authentication so that an
   952→        unauthenticated client on a local-only server can't inject itself
   953→        into another user's long-term memory scope by guessing a key.
   954→        """
   955→        raw = request.headers.get("X-Hermes-Session-Key", "").strip()
   956→        if not raw:
   957→            return None, None
   958→
   959→        if not self._api_key:
   960→            logger.warning(
   961→                "X-Hermes-Session-Key rejected: no API key configured. "
   962→                "Set API_SERVER_KEY to enable long-term memory scoping."
   963→            )
   964→            return None, web.json_response(
   965→                _openai_error(
   966→                    "X-Hermes-Session-Key requires API key authentication. "
   967→                    "Configure API_SERVER_KEY to enable this feature."
   968→                ),
   969→                status=403,
   970→            )
   971→
   972→        # Reject control characters that could enable header injection on
   973→        # the echo path.
   974→        if re.search(r'[\r\n\x00]', raw):
   975→            return None, web.json_response(
   976→                {"error": {"message": "Invalid session key", "type": "invalid_request_error"}},
   977→                status=400,
   978→            )
   979→
   980→        if len(raw) > self._MAX_SESSION_HEADER_LEN:
   981→            return None, web.json_response(
   982→                {"error": {"message": "Session key too long", "type": "invalid_request_error"}},
   983→                status=400,
   984→            )
   985→
   986→        return raw, None

  1391→    async def _handle_create_session(self, request: "web.Request") -> "web.Response":

  1497→    async def _handle_fork_session(self, request: "web.Request") -> "web.Response":
  1498→        """POST /api/sessions/{session_id}/fork — branch via current SessionDB primitives."""
  1499→        auth_err = self._check_auth(request)
  1500→        if auth_err:
  1501→            return auth_err
  1502→        source_id = request.match_info["session_id"]
  1503→        source, err = self._get_existing_session_or_404(source_id)
  1504→        if err:
  1505→            return err

  1507→        if err:

  1510→        fork_id = str(body.get("id") or body.get("session_id") or f"api_{int(time.time())}_{uuid.uuid4().hex[:8]}").strip()

  1519→        # than inventing a parallel fork store.

  1529→        db.replace_messages(fork_id, messages)
  1530→        title = body.get("title")
